The JavaScript

function saveStory(){ var name = document.getElementById("name").value; var quest = document.getElementById("quest").value; localStorage.setItem(name, quest); //This will call a local storage object that will save onto the local drive alert("Successfully Saved Story"); } function loadStory(){ var name = document.getElementById("name").value; var HTML = localStorage.getItem(name); //This will call local storage object that will load the file based on a key value pair if one exists document.getElementById("display").innerHTML = HTML; }
